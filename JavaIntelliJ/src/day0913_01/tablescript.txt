DROP TABLE ordergoods;
CREATE TABLE ordergoods (
	seq					int,
	goodscode 			varchar2(20),
	goodsname 			varchar2(30),
	rawmaterialcode 	varchar2(20),
	rawmaterialQty		int,
	insales				char(1)
); 
-- 기본키: seq
-- 기본값: insales -> 'n'
